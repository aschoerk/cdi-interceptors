22  /**
23   * The Class LoggingInterceptor add a special kind of logging advice to classes / methods.
24   * Calls with parameter names, results and exceptions are logged in a single line.
25   *
26   * A {@link BytecodeReadingParanamer} is use to detect parameter names.
27   * If paranamer does not deliver names, a default of arg0...n is used.
28   * 
29   * @see Logging
30   */
31  @Interceptor
32  @Logging
33  public class LoggingInterceptor {
34  
35      /** The paranamer used to get parameter names. */
36      private Paranamer paranamer = new CachingParanamer(
37  	    new AdaptiveParanamer(
38  		    new BytecodeReadingParanamer(), 
39  		    new PositionalParanamer()
40  	    ));
41      
42      @Inject
43      private StringTransformer stringTransformer;
44  
45      /**
46       * Logging method invoke.
47       *
48       * @param ctx the invocation context
49       * @return the result object
50       */
51      @AroundInvoke
52      public Object loggingMethodInvoke(InvocationContext ctx) throws Exception {
53          Logger logger = LoggerFactory.getLogger(ctx.getTarget().getClass().getName());
54          Logging annotation = getAnnotation(ctx.getMethod(), Logging.class);
55          Level level = annotation.standardLogLevel();
56          int maximumCount = annotation.logItemLimit();
57          boolean measureDuration = annotation.measureDuration();
58  
59          StringBuilder sb;
60          long start = currentTimeMillis(measureDuration);
61          long duration;
62          Object result;
63          try {
64              result = ctx.proceed();
65              if (logger.isEnabled(level)) {
66                  duration = currentTimeMillis(measureDuration) - start;
67                  sb = getCallString(ctx, maximumCount);
68                  sb.append(" returns ");
69                  stringTransformer.transform(sb, result, maximumCount);
70                  appendDuration(sb, measureDuration, duration);
71                  logger.log(level, sb.toString());
72              }
73          } catch (Exception e) {
74              if (e instanceof RuntimeException) {
75                  level = annotation.severeExceptionLogLevel(); 
76              }
77              if (logger.isEnabled(level)) {
78                  duration = currentTimeMillis(measureDuration) - start;
79                  sb = getCallString(ctx, maximumCount);
80                  sb.append(" caused ");
81                  sb.append(e.getMessage());
82                  appendDuration(sb, measureDuration, duration);
83                  logger.log(level, sb.toString(), e);
84              }
85              throw e;
86          }
87          return result;
88      }
89  
90      private long currentTimeMillis(boolean measureDuration) {
91          return measureDuration ? System.currentTimeMillis() : 0;
92      }
93  
94      private <T extends Annotation> T getAnnotation(Method m, Class<T> clz) {
95          // first look at method-level annotations, since they take priority
96          for (Annotation a : m.getAnnotations()) {
97              if (clz.isInstance(a)) {
98                  return clz.cast(a);
99              }
100         }
101         for (Annotation a : m.getDeclaringClass().getAnnotations()) {
102             if (clz.isInstance(a)) {
103                 return clz.cast(a);
104             }
105         }
106         throw new RuntimeException("@" + clz.getName() + " not found on method " + m.getName() + " or its class "
107                 + m.getClass().getName());
108     }
109 
110     private StringBuilder getCallString(InvocationContext ctx, int maximumCount) {
111         StringBuilder sb = new StringBuilder();
112         sb.append("call ");
113         sb.append(ctx.getMethod().getName());
114         sb.append("(");
115         Object[] parameters = ctx.getParameters();
116         String[] parameterNames = paranamer.lookupParameterNames(ctx.getMethod(), false);
117         for (int i = 0; i < parameters.length; i++) {
118             sb.append(parameterNames[i]);
119             sb.append("=");
120             stringTransformer.transform(sb, parameters[i], maximumCount);
121             if (i < parameterNames.length - 1) {
122                 sb.append(", ");
123             }
124         }
125         sb.append(")");
126         return sb;
127     }
128 
129     private void appendDuration(StringBuilder sb, boolean measureDuration, long duration) {
130         if (measureDuration) {
131             sb.append(" [duration=");
132             sb.append(duration);
133             sb.append("ms]");
134         }
135     }
